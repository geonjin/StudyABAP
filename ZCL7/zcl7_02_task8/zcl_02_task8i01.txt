MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'CANCEL'.
      CLEAR ok_code.
      LEAVE TO SCREEN 100.

    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE check_sfliht INPUT.
  SELECT SINGLE *
    FROM SFLIGHT
    INTO CORRESPONDING FIELDS OF WA_S_FLIGHT
    WHERE CARRID = SFLIGHT-CARRID
      AND CONNID = SFLIGHT-CONNID.



ENDMODULE.
