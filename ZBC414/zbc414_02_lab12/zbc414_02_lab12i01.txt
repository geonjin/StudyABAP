MODULE user_command_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    when 'SEARCH'.
      CALL SCREEN 200.

  ENDCASE.
ENDMODULE.

MODULE exit INPUT.
  cASE ok_code.
    WHEN 'CANCEL'.
      LEAVE TO SCREEN 100.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE check_id INPUT.

  SElect SINGLE * from scustom
    into CORRESPONDING FIELDS OF WS_CUST
    WHERE id = gv_id.

ENDMODULE.
